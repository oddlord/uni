\newboolean{trivialAnswers}
\setboolean{trivialAnswers}{true}
\newcommand{\question}[1]{%
  \quad%
  \begin{wrapfigure}{l}{0.1\textwidth}%
    \vspace{-30pt}%
    \begin{center}%
      \includegraphics[scale=1]{hand.jpg}%
    \end{center}%
    \vspace{-30pt}%
  \end{wrapfigure}%
  \noindent #1%
  \FloatBarrier
}%
\newcommand{\answer}[1]{%
  \vspace{5pt}\par\noindent
  \textbf{\underline{Answer:}}%
  \begin{addmargin}[1em]{1em}% 1em left, 2em right
    #1%
    \FloatBarrier
    \begin{flushright}%
      \qedsymbol%
    \end{flushright}%
  \end{addmargin}%
  \vspace{10pt}
}%
\newcommand{\trivialanswer}[1]{%
  \ifthenelse{\boolean{trivialAnswers}}{%
    \answer{#1}%
  }{}%
}%
